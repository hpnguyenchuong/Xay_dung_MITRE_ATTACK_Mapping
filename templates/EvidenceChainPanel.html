'&' : '?'}t=${new Date().getTime()}`) .then(res => res.json()) .then(data => { console.log("Evidence Chain Data:", data); setChainData(data.evidence_chain || []); setLoading(false); }) .catch(err => { console.error(err); setLoading(false); }); }, [selectedDrone]); if (loading) { return
; } return (
link EVIDENCE CORRELATION CHAIN
{chainData.length === 0 ? (

) : (

{chain.raw_packet}
arrow_forward

{chain.decoded_json}
arrow_forward
L3: Artifact
{chain.artifact}

{chain.rule_trigger ||"heuristic_match"}

{chain.technique ||"UNMAPPED"}
arrow_forward

{chain.confidence}

{chain.ics_translation}

{chain.impact}
))}
)}
); };
