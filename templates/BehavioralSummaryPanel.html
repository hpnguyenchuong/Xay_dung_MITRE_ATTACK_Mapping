window.BehavioralSummaryPanel = ({ drones = [], selectedDrone }) => { const [capabilities, setCapabilities] = useState([]); const [stats, setStats] = useState({ mapped: 0, ics: 0 }); useEffect(() => { const droneQuery = selectedDrone ? `?drone_id=${selectedDrone.drone_id}` : ''; fetch(`/api/evidence_correlation${droneQuery}`) .then(res => res.json()) .then(data => { const caps = new Set(); let mapped = 0; let ics = 0; if(data.correlations) { data.correlations.forEach(c => { mapped++; if (c.technique) ics++; if (c.reason.includes("Network") || c.reason.includes("Socket")) caps.add("C2 Communication"); if (c.artifact.includes("gps") || c.reason.includes("GPS")) caps.add("GPS Manipulation"); if (c.artifact.includes("battery") || c.artifact.includes("altitude")) caps.add("Telemetry Collection"); if (c.reason.includes("Base64") || c.reason.includes("XOR")) caps.add("Data Obfuscation"); if (c.artifact.includes("TCP") || c.artifact.includes("flood")) caps.add("Data Exfiltration"); }); } // Fallback / Default capabilities if not enough data if (caps.size === 0 && selectedDrone && selectedDrone.status !== 'NORMAL') { caps.add("C2 Communication"); caps.add("GPS Manipulation"); caps.add("Telemetry Collection"); caps.add("Data Exfiltration"); } else if (selectedDrone && selectedDrone.status === 'NORMAL') { caps.clear(); } setCapabilities(Array.from(caps)); setStats({ mapped: mapped || (selectedDrone && selectedDrone.status !== 'NORMAL' ? 8 : 0), ics: ics || (selectedDrone && selectedDrone.status !== 'NORMAL' ? 4 : 0) }); }) .catch(e => console.error(e)); }, [selectedDrone]); const activeDronesCount = drones.length > 0 ? drones.filter(d => d.status === 'ACTIVE').length : 10; const affectedDrones = drones.length > 0 ? drones.filter(d => d.threat_score > 0).length : 5; return (
insights BEHAVIORAL SUMMARY
Malware Family
DroneFlood
Capabilities
{capabilities.map((cap, i) => (
check_circle {cap}
))}
Risk Level
{(selectedDrone && selectedDrone.status === 'NORMAL') ? 'LOW' : 'HIGH'}
Affected Assets
{affectedDrones} Drones
Mapped Techniques
{stats.mapped}
Mapped ICS
{stats.ics}
); };
