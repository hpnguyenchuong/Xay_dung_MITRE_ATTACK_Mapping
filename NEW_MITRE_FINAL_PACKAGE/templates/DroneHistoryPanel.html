window.DroneHistoryPanel = () => { const [historyList, setHistoryList] = React.useState([]); const [selectedHistoryDrone, setSelectedHistoryDrone] = React.useState(null); const [timelineEvents, setTimelineEvents] = React.useState([]); React.useEffect(() => { const fetchHistory = () => { fetch('/api/threat_score') .then(res => res.json()) .then(data => { if(data.rankings) { setHistoryList(data.rankings); } }) .catch(err => console.error("Error fetching history:", err)); }; fetchHistory(); const interval = setInterval(fetchHistory, 5000); // Poll every 5s return () => clearInterval(interval); }, []); const handleSelectDrone = (droneId) => { if(selectedHistoryDrone === droneId) { setSelectedHistoryDrone(null); setTimelineEvents([]); return; } setSelectedHistoryDrone(droneId); fetch(`/api/timeline?drone_id=${droneId}`) .then(res => res.json()) .then(data => { if(data.timeline) setTimelineEvents(data.timeline); }) .catch(err => console.error("Error fetching timeline:", err)); }; return (
{/* Header */}

Historical Connections

{historyList.length} Records
{/* Content */}
{/* Left side: List of historical drones */}
{historyList.map(drone => (
handleSelectDrone(drone.drone_id)} className={`p-3 border-b border-slate-700 cursor-pointer transition-colors ${selectedHistoryDrone === drone.drone_id ? 'bg-[#3b82f6]/20 border-l-2 border-l-[#3b82f6]' : 'hover:bg-slate-700/50'}`} >
{drone.drone_id} 0 ? 'bg-transparent text-[#f43f5e]' : 'bg-[#10b981]/20 text-[#10b981]'}`}> Risk: {drone.score}
Last seen: {drone.last_updated}
))} {historyList.length === 0 && (
No historical data found.
)}
{/* Right side: Timeline details */}
{!selectedHistoryDrone ? (
Select a drone to view attack timeline
) : (
▶ Event Timeline: {selectedHistoryDrone}
{timelineEvents.length === 0 ? (
No events recorded for this session.
) : ( timelineEvents.map(event => (
{event.event_type} {event.timestamp.split(' ')[1]}
{event.message}
)) )}
)}
); };
